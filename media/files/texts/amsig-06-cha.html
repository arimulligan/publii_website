آڨلّان غرسن.

آيهْبّاب ن ـ وَخّام سقصانِيهَنْ مَانِيسْ آدُوسِينْ اُو ماني آذ ـ رُوحَنْ

اَذْ ـ يوذان آڨلّان اَمسَلْقان اِيذْسَنْ.

ينّأ: " اَمبصّحْ اَلّانت ثلاثة  ن ـ تمحانجِيّين  اُونّدجّيمشْ آهَن ْـ تنَّسَّرڨ،"

" واڨ ـ اَتَّبِّين اِيصغاڕَنْ يقّار بلّي  يلّا ويشت يَخْلَقْ اَدِّيمُّث آيّا ثَبهَا اُو ثَحْلَا.

آمْــڔابَـضْ يَقّارْ هضْرا ـ اَثغاوسَا  اِي اَدّيمُّث اَهْفِيَّا لَبهاوَثْ ذَ ت تّْمُوحْلِي اَنّسْ.

اُو ثامغاڕث  ثَقَّارْ بَلّي اُور ـ يَقِّيمشْ لاكَدْ لَامان ذِي اَدّيمّثثْ

اُو مَشتا وَرنَا ـ ن ت ثْمَتَّانْت، اُورـ ثَتندجّامش آتّضمَن بَلّي آنَّـافْ ثامسدوغَث."

 ينّا اُوـ عــزَّاب " نتَّشتَاقاي ڨاع اِي لَهْنا، (لامان)،

" اَمبصّح اُوهَنّتَّاف لاكد ذڨ ـ اُوموكان  لَا ذي اَدّيمّث آيّا، لَا ذي ـ لاخرث.

ــ خاطراش اُوـ نَسّينشَا آنڕاضا رَبّي."

سُـڨا يَسْلَا آمـسيڨـل ثوثلايث آيّا، ينّا،

" ثامغاڕث، آنَغبُونْتْ، هُڨَّذ  آورَاـ تِيقَبّلشَا رَبّي اَخلاصْ،

خاطراش اُورـ هَتندجّامش آتّيڨ آڨ ـ طَلّب  فلّانغ  الدِّينْ."

ثنّا ثامَطُّوث ـ ن ـ اُو ـ عــزَّاب:

" نشنين بَعد  بَعْدْ اُورـ نتدندجّامشْ آنّيڨ  آڨ أ اَطْلَبْ، آَخْلاصْ."

" نشنين نسدّغ ذڨ موكانن  ذي ـ بَرِّيَّنْ، ينّا آرڨاز اَنّس،

" اُورـ نَسّالشا اِي لَاذانْ، لَا  ذي اَنّصافْ ن ـ واسّ، لا ذُڨ آمسّايْ.

اُوغَرنغش لخبر ذَڨ ـ  يُورْن  اَذْ ـ وُوسّان نـوژومي.

ولّيش نِ ـ طَلّابَن آڨَلّان آندجّم آس ـ نّنُّوش اَصّاداقَث.

اُورـ نسّينش  آَمك سَبّْغَن لُوضُو.

الحجّْ  ذَ ـ الطّالب ن ـ الجامع ڨاع بَعْذَنْ فَلّانَغْ.

اُوـ نسّينش ثوثلايِيين آڨلّان اَتْواقْڔايَنتْ ذي الجامع، اَرْنِي،

اُونْفهّمشا اَلْمَعنا اَنّْسَنْتْ. ماتّا آغ ـ نْيَضْرَانْ اَمّالَا،

مُكشا آنّيلي نَحْلى ذي ـ هِيطّ ـ نْ ـ يِيلّو؟

اُو مُكشا آندجّم آنّيلي ذي ـ لَامان غَرسْ؟"

 اِيسقصا آمسيڨل: " ولّيش ن ـ الطّريقَث هيشت باش آسْ ـ نَّعْجاب اِي ربِّي

وآنّاف لامان انيغ لَهنا؟

 يَنّا اُو عَزّابْ: " اِيّا ـ ڨَلّان اَتروحَان  غَرْ الجامَعْ بَرْكْ ذنَّهنين آڨلّان آهَنْ يَقْبَلْ رَبّي

وآذ ـ عاشن ذي ـ لَامَانْ،"

آرڨاز آَنّس يمّضرَان غرسَن اُو يَنّا، " هاتّايَا ثاقصيصث اتعاواذينيت اِيمَزوُورَا

 زيك يلّا  شْرا ـ نُ ـ اُوڨَلّيذ، لَخْطَرث اِيعَرضَدْ آقَضْعيث ـ نْ ـ يَرڨازَنْ باش آث ـ عشّان اِيذَسْ ذي لَقْصَرْ.

اُوذفَند غرس

اوُذْرَنْ اَزداثَس اَحطَاڔمِينيث لاباس.

اُورْـ اَسْوِينْ اَدّخّان اُوـ اَسُّوفِيسَنْ، اُورـ نَدجِّيمنشْ

آذرفذَن هيطّاوين  غر ـ  ثيخذيمين ـ نْ ـ لَقْصَرْ،

انّيغ  مَشتا ذاوال اُوذيحْلينشْ اورـ يَنّـُوسْتَــيــفـــش جار ن ـ اَشّْواربْ اَنّْسَنْ.

بَانَنْ بَلّي  ذي ـ اَرْڨازَن ذي مَثْرَبّْيَنْ

وآڨلّيذ يعجاباس الحال

لامكنّاتي، نتّا ذارڨاز ذوفهيم اُو يَسَّنْ آمكْ اَڨٌّورن يوذان.

سُڨّا روحَن اِينيجِيوَنْ،

يَصْرَف اِيخذيمن اَنَسْ آذ ـ سقصان اِيمَعشارْ اَنّسَنْ

أمك اَتّيـِڨَنْ ذي ـ اَسُّوق

وآدْ ـ ياوي لخبار ـ سْ ـ غَرْ الْخَلَاثْ اَنّسَـنْ  ـ فْ ـ آمْك آڨُّـورن ذڨّْ آخّام

 سُڨا اُوذفَندْ ايخذيمن غَر ـ لَقْصَرْ، اَويند لخبار ڨاع اِيْ ـ اُو ڨَلِّيذْ،

يُوذان آيّا اَتشّاثَنْ الْخَالَاثْ اَنّْسَنْ؛  واَتْكلّاحَن ْ لَعْبَاذْ ذِي ـ اَسّوقْ

اَتّمنوغَن اِيذْ ـ نِ ـ اِيمَعْشارْ اَنّْسَنْ. كُلّ اِيشتْ يَتشّاثْ ـ ف ـ اُوغيڔْ اَنّسْ

اُورْـ اَسْتاهِيلَنشَا باش آذ ـ ولّان ذي ـ مـدُّوكَالْ ـ اَنّكْ.‘

 ينّاسنْ آڨَلِّيذ، " اُور ـ اَخْسَغْشْ آذرنِيغ آهْنَڙڕَغ ذي لَقْصَرْ اِينُوغْ!‘"

واِيكمَّل آمعزَّب اُوـ ينَّـا: ’’ ذي ـ الْجامَعْ،  يُوذان اَتّاذْرَنْ ازْداثْ ـ ن اُوڨَلّيذ آمُقرَانْ،

وآذيڨَن كُلّشي باش آذسّنْعْثَنْ بَلِّي آهَنْ اَحْلَان.

اَمبَصَّحْ آلدا آذَرْڨَن ـ سِي ـ اَلْجامَعْ،  ذينِين آڨَـلِّيذ آهَنْ ـ اِيعُوٍّسّْ

آمكْ آمّوُڨَنْ.  ماتّا هَنِّيذ؟‘‘

مَا يَلّا اَثْژالِّيطّْ ذو ژوميْ  اُور ـ اَتُّــوعانتشْ لَعْبَذْ يحلَا

اِيذ ـ ن ـ لُمّالي اَنّس، هِيماتّا ـ ن ـ لْقيمَتْ آغَرْسَنْ؟

اَذ ْـ وِيشتْ آڨلّان يتكلّاح يُوذان  ذي السُّوق،

مُكْشا آذي ـ نَدجّم آذ ـ اِيراضَا رَبّي ذي اَلْجَامَعْ؟

ف ـ ثايّا ولّيش ـ نْ ـ لَامان ذي اَدِّيمُّث آيّا

خاطراش اُورـ نسِّينش مُكشَـا آنّراضا ربّي  سَ ـ اَسِّـيـرْثْ اِيلِـيْقَـنْ اَذْ ـ تِـيكْلي ـ ن ـ ڨَـدّْ ڨَـدّْ.

اُو ف ـ لْعُوض ـ ن ـ وَامّا، اتحاوالَـن آذيـلين اَحلان ذي ـ ثِيـطّاوين اَنّسْ  

قرّان آوال ن ربّـي، اطزومن رمضان!

وَالْفَنْ ـ س ـ اَثْغاوسِيوينْ آمَك اَلّانتْ  اُوذْ ـ ذروزِّينشا اَفْ ـ وَجْدِيــذْ!"

يُـوقَّلْ غَرْسَن آمـسيڨـل، اُو يَنّا، " ثايّا اُور ـ اغْرَسّْشا لَاكَدْ الْمَعْنَتْ

آقّلت الًــصِّيغَـثْ ـ ن ـ وَژڕَفْ يَالّا.

يُوذان ڨاع اَقَّارَنْ بَلّي آژڔَفْ  اَلْقِيمْتْ اَنَّسْ اَتّامُقرَانتْ لَاباسْ  اُوْ ـ يَتْنُوزْ يَغْلَا.

اَمبَصّحْ اِيْ ـ مَاتّا اِيصَلّحْ؟ اَمّاغَفْ أ آخَذمِي ـ ن ـ لِينوكْسْ ولِّيشْ ذِيسْ اَلْفَـايَتّْ؟

آندجّم آنّبِّي ـ اِيــسّْ اَلْخَردلْ. اَنِّــيغ آنفثاح اَن ـ وُوزّال

آندجّم آنّفذَك اِيــسّْ الْباب.

سَ ـ تْغنـجَايَـثْ نُوٌقَـشّوظ آندجّم  آنتشَّــارْ ثِـيڔْبّا اَسْ ـ تَرْوَايَـثْ.

ثِـيغَاوْسِيوينْ آيّا ڨاع نتّـافِيهنت. اَمبصّح  اَثخاثَمتْ اَن ـ وَڙڕَفْ، اِي مَاتّا اَثْصَلّحْ؟

اَنِّـيغ آخَلخَـالتْ ـ ن ـ وَژڕَفْ، مَاتّا اَلْفَايَتّْ اَنّـسْ؟

يَتْـبانايَد يُوذان اُورـ اَسِّـينَـنْشَا واڨّلّان ـ ذيس الْفَايَـتّْ

سُڨْ ـ آڨَلّان ولّيش ذيسْ اَلْفايَتّ! آعزّاب يَقبَل ثوثلايث آيّا

" وآهَنْ وَالْفَنْ اَتّيڨَن  آمَك اَتِّـيڨَنْ يُوذان ڨاع،"

" اَتّيڨَن آمْك آتّـيڨَن اِيّا آڨ ـ عَدّان،

اَتّبّاعَنْ لَعْوَايذْ ـ ن ـ لمالي انّسن سِي باباثسَن غَرْ مَمِّـيثْسَنْ

اَبْلَا مَا هاذْ سَقْصان بَلّي  ذيسنت الفايتّْ اَنِّيغ آراح."

هَثّْـنَهّذ ثامَطُّــوثْ اَس ـ لاباس اُو  ثَـنّا: "وايّا ڨاع يسّبحَــارْ،".

" ذي الرّاي اَنّك ولِّيـشّا ن ـ وَبريذ اِيشت ابْـلَا وايّا ـ آڨ ـ تّعجابن اِي رَبّي؟

آرڨاز اَنّس يَـنّا، " ثافوناسَتْ اَهضَوّلْ ـ سْ ـ وَسْغُــون ذي اَثْڕَعقيثْ

ثَـقّيـم ثَتّـنّضْ ذڨّ اِيشْتْ ـ ن ـ اُوموكان اَثرفَّسْ ذي ـ اثْمُورْثْ اَمْغِيـرْ اُورـ ذِيـقّيمِشَا لاكد اَقْلِي سِيْ اَلْمَرثَعْ.

اُورـ ثَحْـبيسْـشَا سُڨّْ اَجْبَاذْ ـ ن ـ وَسغُونْ اُوْ هَتِّـيڨْ كُلّشِي، آت ـ ثَخْـسْ آتّـبِّـيثْ

اَنّـيغ آتَّـقْلَع مِيغِـيسْ باش آتّخلَضْ غَرْ لَحْـشيش يَحْلَانْ.

هَامّا آنَلّا نَـشْـنِيـــن.

نشنين نتّشتاقاي  آنّروح  باش آنّاف ثَاڨَلْذا ـ ن ـ يِيلُّو،

اَمبَصّح نشنين نَـنْڊَض ذِي لَعْوَايَـذ،  اُو نَـتَّاڨّاذ ماتّا آذينِــين فَلّانَغْ يُوذانْ!"

يَنّا آمسيڨَال: " هنّيذانَغ آـ مَاسْ بلّي اَثغاوسَا آڨ ـ اَتْهِيــمّان، نتّاث آتِّــيڨم واڨّْ ـ اَحْلَان اِيْ ـ يُوذان،"

، " اَمبصّح آمَك آنَّـسَّن واڨ ـ اَحْلان؟

ثابَثْ اَلّانت ثيغاوسِـيوِين اَحْلَانت ـ اِي وَاييَّـا، اُو ـ اُورـ حْلــينتشْ اِي وِينِّي،

اُو ـ ثِـيغاوسِـيوِينْ اُورَـ اَحْلِـينتْشْ ـ اِي وايَّـا، واَحْلَانتْ اِيْ وِينِّي."

ثامَطّـوثْ ـ ن ـ اُوعزّاب ثَبّاسْ هُوثْلَايَثْ

" اُورـ اَفهِـيمَغشْ ماتّا ثَخسَذ آتِّـينِيذ، آ آمـسيڨـل!"

اِيْ ـ هيه ، مايَلّا وَلْتْمَـا  ذَ ـ نَّـتشْ غرنَغ ثاتَفّاحَث

وآتاويغ نتش ڨاع، ثايّا ثحلَا اِي نتش، امبَصَّح اُوثحليش اِي ـ نَتّــاثْ.

ما يَلّا اَبّيغت اُو اُوشيغاس آزڨَنْ، آنتش  اَسَّـنقصَغْ سُڨ ـ آڨِلْ اِينُوغ اَملِيحْ

واَرنيغاس اِي ـ نتّاثْ.

يَعني آندجّم آنّبضااَثمُوحْـلِي جاراثْنَغْ؟

مَا يلّا اِيروح وِيشتْ غر ـ السُّوق

ويَسغا اَثـغَاوسا ثَغلَا  ـ ف ـ اَسُّومْتْ اَنّسْ،

ذِي لْبَالْ اَنَّسْ نتّا يَتُّوكَــلّحْ

اَمبصّح ذي لْبال  ـ ن ـ واس ـ اِيـزّنْزَنْ، ذ ـ لْفَايَتّْ

اَنّيغ آمتشّوك آڨلّان اِيسَرّح ـ سَ ـ اّدّوبان آلْدا ـ آذ ـ يَسْلْ اِيْ لَاذَانْ

آذ ـ يَدجّ اَدُّوبَان  اِيرتَّع ذي ـ اَثعَڔقُوبَثْ  باش آذ ـ اِيروح غَرْ الجامَع،

آنِّـيني بَلّي اَثْـغَاوسا ثَحْلَا،

اَمبَصّح ـ ف ـ باب ـ نَ ـ اَدّوبان، اَتّْـغاوسَا اُورـ ثَحْـلِيشْ.

اُورـ تندجّامَغشا آذافَغْ الحلّْ اِي لْمُوشكِيلْتْ آيّا.

آمكْ آندجّم آنّژڔْ واڨّْ ـ اَحْلَان سُڨ ـ آڨلّان اُورـ ذي ـ احْليشَا؟"

 ينّا آمـسيڨـل:" اغْرِي آسقصي وِيشْتْ، ".

" مَا يلّا اِيڨِـيغ اَث ـ غاوسَا سَ اَنَّـيَّـثْ يَـحْلَان،

واَمبَعْد آذافَغْ بَلِّي  اَثْـغاوسا آيَّا هتضوڔَّاي وِيشْتْ اَسّيُونِّي،

مُكِشَا آذ ـ ندجّمَغ آذ ـ رانْجِيغ ثامسالتْ؟

مُكشا آذ ـ ندجّمغ آذعذلَغ لْحالتْ اِيذ ـ ن ـ يُوذان آڨلّان ضوڔَغ واِيذ ـ ن ـ رَبّي؟

ما يَلّا طَيّشَغْ آحْذيــرْ  باش آذ ـ نَهڔَغْ اِيزمارَنْ اُورـ اتْبعّاذَنشْ،

اَمبصّح لُوكان آتّلْقَف آمتّتشوك وآذـ يَمَّثْ،

ماتّا آذيڨغ باش آي ـ سَمَحْنْ لُمّـاَلِي اَنّسْ؟

اُو ماتَّا آذيڨغ باش آي ـ يَسْمَحْ رَبّي؟

اُو لوكان آذ ـ اَمْنُوغَغْ اِيذ ـ ن ـ ويشت وآثوثغ س ـ لْبونْيَا،

اَمْبَعدْ آذافَغْ بَلِّي يَلّا غَرْسْ  الْحَقّْ ـ ف ـ نَڨَّارو،

ماتّا آذيڨغْ باش آيّسمَحْ ـ نتَّـا؟ اُوـ ماتّا آذيڨَغ باش آيَّسمَح رَبّي؟

اَقّيمَنْ ذينين  ثاسمانتْ اِيذ ـ ن ـ اِيعزَّابن،

اِيمرذينك آمـسيڨـل يَژڊَا فالُو  سِي اَسّْـعَفْ ـ ن ـ تَزْدايْثْ

واِيضُوبّْ ثِيطَرباقين ـ نَ ـ اَدّرَارِي ڨاع.

اثزيزوتّْ ثَڨناسَن وَاَهْضُوبّاسَنْ اِيكَتَّافَنْ اَنّسَنْ ـ اَسْ ـ تَسَّڨْنِيثْ ذوـ فالُو

آڨلّان يوشاسْهَن واڨ ـ اَتّـبّـين  اِيصـغاڕَنْ.

واَمبَعْد آمـسيڨـل اَذ ـ اثزيزوتّْ اَدجّينيهَنْ اُوْ كَمّْلَن آبْرِيــذْ اَنّسَنْ،

اُو نَهنِين اَتْمَكثَايَــن اِيسَقصايْ ڨاعْ ن ـ اُوعزّاب ذ ـ اَتّمَطُّوثْ اَنّسْ.

يوقّا الطَّرف ويس سَتّة سي ثَقصيصتْ
