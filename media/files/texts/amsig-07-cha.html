آمـسيڨـل اَذ ـ  ولتماس ارنين ـ تَـبّعَن آبريذ ـ اَنّْــسَـنْ ـ اَمغيـر آنّصاف ـ ن ـ وَاسّْ،  

اُو نهنين اڨُّــورن انّاك آهناين سَنّ ـ ن ـ لَغرُوز

اَلّان ذ ـ اِيْـمْرَوّْحَنْ سِي ـ اَثرعقاي،  الݒالَا ـ فْ ـ واژي.

اَڨرِين فلّاسن اَسّْـلام ـ  وُ اُوڨِـيرَن اِيذ ـ جميع ذڨّ اَبريــذْ.

اُوثلايَن اقْـلِي  ـ فْ ـ اَثْ ـ كَارْزَا ـ ذ ـ اتّْـمَا ڨـرَا،

اَمْـبَعدْ آمـسيڨـل ايسقصا، "  آمكْ اَقّارِينـَاوَنْ؟"

ينّا اِيشْتْ اَسِّــيـسَنْ، " نتش ـ اَذْ ـ حَمُّو بُـوـ ثْـمَـارْثْ،

اُوـ نَقَّـارَاسْ اِيْ وَايّْ حَمُّو آعنّاب. " آمـسيڨـل يَبْـحَرْ او ينَّـا:

" آنتش اَژڕيغ اَمَّاغَف اَقّاريناك ’ بُو ـ ثْمَـارْثْ‘، " خاطْرَاش غَرْكْ اَثْمَارثْ

اَمبَصَّحْ اَمّاغَف اِسْـمْ ـ ن ـ وْعنّابْ؟" " اِي ـ هِيه اِيْواجّبْدْ وِيــنِّي ـ سُـو غُــوشِّـي

" اَشْرَا ـ ن ـ الْخَطْرَثْ اُوكْـرَغْ ـ اَثْ ـ ڨَـاژيطّْ اِيْ اَلقَايـــدْ ـ ن ـ اُوفِـيلَاجْ واَمْبَعـد اُوفِينِيثْ اَغْـرِيْ."

آمـسيڨـل، يَبْـحَرْ، وِ ـ اِيسَـقصاثْ، " مَلْمِي وَا ؟"

 يَنّا: " عَـامِـين،" " اُو سِي ـ اَلْوَقَّثْ اِيــنِينْ، اُو ـ ذوكِيرَغْشْ اَخْلَاصْ."

"آيْوَاه،" يَنّا آمـسيڨـل، " اِيمِيرا اُوهَوَلِّـيذشَا ذاعَنـابْ!"  

اَمبصَّح حَمُّو بوـ ثْمارثْ يَسِّـيوَلْ او ينّـا: " واڨَلّان ذاـ مَشبُوهْ،" " ديمَا اَتشوكّان ذيس!"

آمـسيڨـل اِيخَمَّـمْ ذيس اَزداث ـ ن ـ وَاهاذْيِـيني، " غَرْك اَثْمَـارْثْ

آتندجّمذ آتّـكسَت مَلْمـي آثَـخْسَذْ وآتبدّلَذ اَزِّيّْ اَنّـكْ.

وَآمَدُّوكـلْ اَنّـك اُورـ اِيندجّامش آذـاِيبّـدّلْ ازيّ اَنّسْ، لاكد نتّا؟"

آه، ينّا حَمُّو بو ـ ثْمَارْثْ، " اِيزاوَن ـ ن ـ لَعبَذْ غَـمِّـينْ ـ فْ ـ ݒَڕَا،

لامكنّاتي الطّْبِيعَث ـ ن ـ تَڮُّوڔضا ثلّا ذاخَلْ اَنَّسْ."

يَنّا آمـسيڨـل.: اَمبَصَّحْ آندجّـم آن ـ بدّلْ  

واڨلّان ذاخَلْ اَنّغ اَنِّيغ آرَاحْ؟"

 يَنّا حَمُّـو بوـ ثْمَارْثْ: "وايّا،"

" ذيس الطّْبِيعَثْ نُومَاكَارْ آتْ هلّا ذيس سِيمَا ـ ڨَّلّا ذامَزْيَانْ ـ

يـَطّْـضِيثْ اِيذْ ـ ن ـ وَاغِي ـ ن ـ يَمّاسْ واِيذْ ـ ن ـ تِيذِي ـ ن ـ التَّعْـب ـ ن ـ باباسْ.

آمتشُّوكْ آثْ آمْ ـ وَصغاڕ ـ ن ـ اُوـ ليلي،

آلْدا آذـ يِـيلِي ذا ـ زِيزَا،  آندجّم آنِّـيڨ اِيسّْ آنَـخْسْ،

اَمبَصّحْ آلْدا آذيِـيلِي يَـقُّوڔ، يَحْـلَا غِـيرْ اِيْ الْعَافِيفْثْ!"

يَمَّضڔانْ آمـسيڨـل غَرْـ اُومَاكَارْ،  اُوْـ يَنّاسْ، " اِيْسَـمْحَاكْ القَايَدْ اَنِّـيغ آرَاحْ؟"

ينّا: " اِيسَمْحَايَا،"  "سُڨّا ـ اُوعِــيـغ ـ اَث ـ ڨاژيـطّْ،

اَمْبَصَّحْ، لَعْمَرْ اَنَّسْ مَا ـ يَتُّو ـ آيْ ـ اِيڨـيغْ.

كُلّْ مَا لَا قِّيغْـثْ يتمَصْخيژ اِيسِّي اُوـ يَتْـسِيبَّايَا،

نَتّا اذْ ڨاع لُمَّالِي اَنَّس."

ينّا آمـسيڨـل: " لامْكَنّاتي لوكان آشْكْ هدجّولِّـيذاسْ بَلِّي  مَعَّاتش آتّاكرَذْ"

" اَمّاغَف اُوـ كَسْـمِيحَنش باش آتّـبذيث ـ فُ ـ اُومَزوَارُو؟

اُوهُوفِـيذْشْ  آمَكْ آتّ اَمرانجيِــذْ اِيـــذْسَنْ؟"

ينّا: " آراحْ!"    " خاطراش اَتشوكّان ذِي دِيمَا.

مَا ـ ثَتْوَاكَــرْ اَثْغَاوسَا ذڨ ـ فِيلَاجْ،

آذـ شُوكّــنْ بَلِّي ذَ ـ نَّتش آڨَلّان اَوِّيــغْتْ!"

آمـسيڨـل يَبحَر اُو ينَّـا: ـ اَمَّـاغـف ـ آ ـ  اَتِّيڨَنْ ـ هامّا

" يُـوذان آيّا،"  ـ  آمَاكارْ يَـنَّـا اُو نتّا اِيْـغوشّْ

آم ـ ثاتَفّــاحَث ثتبانَــدْ ثَـحْلَا  اَسْ ـ پَـرّأ  اُوْـ نَـتّأثْ  ثَرْمــجْ اسْ ـ داخلْ

الشرّْ آڨلّان اِيڨِيغ  اَژڕيـنِـيـث ـ ذينْ ـ ذينْ اَمبَصّحْ آڨلّانْ خَدّمَنْـتْ، نَهـنِينْ

اَسّْــنَنْ آمْـكْ آثْ ـ صَّغْبـڔَنْ.

آكْ ـ عَاوْذَنْ بَلِّي آهَنْ اَحْلَانْ، اَمبَصّحْ نتش اَغْرِي لخْبَرْ بَلِّي فُوحَنْ اكْثَرْ اِينُوغ.

بَرْك،  اَتّيڨِيـنِيهَنتْ اَتُّوفْرِا باش حَدْ اُوـ هنِيـژڕْ."

يَنّا آمـسيڨـل: " غَرْكْ الحَقّْ، "

" يُوذان اُورـ اتْنَحْـشامَنش ـ فْ ـ آيْفْ آتِّيڨَنْ آلدا حّدّْ اُوـ هْنيژڕ."

 يَنّا اُوماكَارْ: " ذِي الْوَقَّث ـ ن ـ لْمَشْتَا،"

" يَتشّاث وَذْفَلْ اُو يَصَّغْـپَـاڔ  لُفـَّاحَاث ڨاع آڨلّان فَرزْعَنْ ذِي ـ اثْمُورْثْ.

اَمبَصَّحْ آلْدا آتّاليد ثافُوكَثْ، اِيْفَـسِّي وَذْفَلْ وآذ ـ يَكْشَفْ بَلِّي لُفّأحَاثْ دِيْمَا ذايَا آلّان.

آذفَل آث آم ـ واڨلّان يَقَّار، ’ آنتش ذارڨاز يَحْلَان

اُوـ اِيڨِيغشْ لَاكَدْ اَث ـ غاوسا اُوثَحْلِيشْ!‘ لَعبَذ آيّا يَسَّرْكُوسْ،

خاطراش يتفّر لُفُوحَثْ ذُڨْ ـ اُولْ اَنَّسْ.

اُو رَبّي آسْ ـ انْيَسَّـنْـعَثْ كُلّشِي آسّْ ـ ن ـ لاخَرثْ،

آمْ ـ اَثْ ـ فُوكَث هسّنـعَاث ثيغاوسيوِينْ آڨلّان اُوـ ذَحْـلِيـنْتـشْ ثِيـغَاوسيوينْ آڨلّان يصّغـپـڕيهَنت فَلّانَغْ وَذفَلْ."

 يَرنِي يَنّا حَـمُّو بُـوـ اثْـمَارْثْ: "عرّمَنْ يُــوذان آڨلّان اَقّارَنْ نشنِيـنْ  نَحْلَا،"

امّبصّحْ ذيسَنْ اَطّْـبِيعَثْ ـ ن ـ تْسَرْكَاسْ، اسّتاغَلّايَنْ  الثِّـيقَثْ ـ ن ـ اِيّيض باش آهَنْ ـ خَذعَنْ.

ذَيـسَن اَطّْـبِيعَثْ ـ ن ـ لْغِــيـرْثْ، اَتّخْـسَنْ  آڨل آهن يخضان,

ذيسَن اَطّـبِيعَثْ ـ ن ـ لَقْبَاحَثْ، اَتّمْـلُوثنْ اِيـذْ ـ ن ـ ڨاع ـ ـ ن ـ اِيَّا ڨَلّان اَتْبَددِّيناسَنْ ذَڨّْ ـ اُوذْمَـاوَنْ اَنّسَنْ.

ذيسَن اَطّـبِيعَثْ ـ ن ـ اَسّْـوَقَّـظ، اَتّخْـسَنْ اَلْخَالَاثْ هِـيِّيض.

ذيسَن اَطّـبِيعَثْ ـ ن ـ اُوفُوخِـي، اّستاغَلَّايَنْ  آيثمَاثْسَنْ ذِ ـ مَدُّوكَالْ اَنّسَنْ ذي ـ باش آذ قضان اثغاوسا انّسن.

ذيسَن اَطّـبِيعَثْ ـ ن ـ اَتّْـفَنْيـِيـنْ، خَلّْـفَن اسْ ـ لُوعوذْ ـ اَنّْسَنْ.

اَلْحَاصُولْ، آنشن نسّنْ ڨَاعْ اِيمَانَّغ آمْك آنلّا!"

يَنّا آمـسيڨـل:" غَرك الْحَقْ،"

" نـسَّن آڨّْ ـ اَتْوافّْرَنْ  ذَڨّْ اُولْ ـ ن ـ لَعْبَذْ.

اَمبصّح اُورـ نَتْـخَسّْش اَخلاصْ  آنّسْمَحْ لْغَلْطَاث ـ ن ـ لعباذ اِييض,

خاطراش نتّشتا ـ آن ـ نَسَّنْعَثْ بّلِي نشنين خِير اَنّســنْ.

آندجّم آن ـ اَصّغــپـڕ اّدِّيفُوّاثْ اَنَّغْ،

اَمبصّحْ مُكْشا آندجّم آنصفّا اِيمانّغ سِي الشرّْ آيّا آڨلّان يَفَّرْ ذِينَغْ؟"

يَنّا حَمُّو بوـ ثْـمارثْ:" ماشي اتّغاوسا ثَسْهَلْ!"

" آقّْـلَت اِيْ ـ اثْفُوناسْـتْ يالّا، ثَقْلَقْ اُورـ ثَتَّـحْبِـيسْـشْ سُڨ ـ حاوَزْ ـ ن ـ يِـيْزان.

هثَتْـهايالْ آلَـقَّـاظ اَنَّـسْ، هتشّأث سِي ـ ضاڔَّن، ثَـتَّـقَّـنْ ثِـيـطَّاوِينْ، ثَـسَّـنْـڨُوڨَأيْ اِخْـفْ،

اَمبصّحْ اُورـ ثندجَّـيمْـِشْ آتّـثْـهَـنّأ سَڨّْ ـ اِيْزَانْ.

اَڨُّـومان آتدجَّنْ ذي الحال انّس.

كُلّْ آسّْ اَتغيمان اَسَّـقْـلَأقَـنْ ذِيــسْ.

هَامّأ ثَسّْعَـدَّايْ  ثافوناسْتْ اُوسَّان اَنَّسْ، اَمغِيرْ آتَّـمَّـثْ.

لاكدْ نشنين كيفْ كيفْ.

ثُوثْلايِينْ فُوحَنتْ ذِي ـ خَـمِّـيمَن اُوذحْـلِـينْشْ اتّاسِـينَانَغْدْ ـ ف ـ اَلْجِـيهَاث ڨَاع.

ااَسَّقْـلَاقِيـنَانَغْ، ـ واسّْ ـ نَزْڨامايـنْــانَغْ واَسَّلهايِينانغْ  او اَسَّـقْـلَاقَـأيِـينَانَغْ.

وايّا اُوغـنـيَتَّـعجَابَايَشْ اَخلاصْ ونَـسّْـحُوسَّايْ بَلِّـي ماشِي ذِي ـ مَرثَـاحَنْ.

نَتحـاوَالْ بْلَا اَلْفايَـتّْ باشْ  آهَنـَنـحَاوَزْ سِي اَلْـجِيـهَاث ڨاعْ،

هَامَّا بَـعْد ثافــونَاسْتْ سُڨّأ اَثْـحَـدجّل ذُڨ ـ لَـقَّاظْ اَنَّسْ

نَـتِّـيڨ آڨَلأّان فـلّانَغ باش  آنّْـحاوَزْثِـيغاوْسـيوِين آڨَلّان اُورـ اَحْـلِينْـتِشْ.

نتژالَّا، نطْـژوم، واَنْـقَڕَّا وَاَنْـحَـفَّـظْ، نتصَڊَّاق،

اَمبصّحْ اُورـ نتندجّامش آنّرْوَل سِي ـ الشرّْ ـ ن ـ اَدِّيمُّثْ آيَّا،

آڨَـلّان يَتّاسَد اُو يَتْـروسْ ـ فْ ـ لَفْـرِيسْـتْ اَنّغْ.

هَامّا نَـسّْـعَدّايْ  اُوٍسَّانْ اَنَّغْ، اَمْـغِيْرْ ثَامَتَّانْتْ.

آهَـنْ ـ اِيـزَانْ اُورـ اَتَّـاذْفَـنشْ  ذاخَلْ ـ ن ـ اَتْـفُونَاسْتْ غِيرْ مَا ثَمُّـوثْ

اَمبصَّـح، ما يَلّا فَلّانَـغ  نَشْـنِـينْ، ثِـيغَـأوْسِـيوِين آيَّأ  اِيشَرّانِـيِّيـن اَتّاذْفَـنتْ ذاخِلْ اَنَّغْ

اُو نَـشْـنِيـنْ وَرْعَاذْ نَـدَّرْ."

 يَنَّا اُومَاكَارْ: " ربِّـي آغ ـ انْيـصـطَرْ!"

" خَاطراش اُوـ نّـتْـندجّامش آنّروَل  سِي اثْغاوسِـوِين اِيفـوحَنت ذِي اَدِّيـمُّثْ آيِّا،

اُورـ نَـتَّـسّـنشْ لَاكْ آ آنَّـكّسْ الْـفُوحَثْ  آڨلّان ذاخَل اَنَّـغْ،"

 ينَّا حَـمُّـو بُوـ ثْمَـارْثْ: "  غَرْكْ الحَقّْ،"

" خاطراش، آمْـكْ نَـنَّا،  لَـعْبــذ آڨَـلّان اُوـ ذيحْـلِيشْ اُوـ ذيتـنـدجّامش آذـ اِيـولّا يَحْـلَا

لَا ذِـ ثِيطَّاوِينْ ـ ن ـ رَبّي، لَا ذِـ ثِيطَّاوِينْ ـ ن ـ لعباذ"

آمـسيڨـل يُوثْـلَايْ

"  آنولّا  نبعذ اخلاص  ف ـ رَبِّي

في ـ مَعْشَارْ، اوـ فيـماننَّغ نشنين بعد.

لُـوكَانْ  لَـهْنا ذُ ـ امْسفهامي بَـرْك  آذـ نَدجّـمَن آذيلينْ جاراثنَغْ!

لُوكان غِيرْ  ـ آت ـ اِيـلِي اَڔحــمَـث

باش آذ اَمّـقڔابَن اِيّا ڨَلّان اَمهايْلَنْ!"

اَقِّـيْـمَـن ڨَاع اَسُّـوسـمَنْ،  اُو ـ اَلْهَانْ ذَڨ ـ خَـمِّـيمَن اَنّـسَنْ.

اَمْبَعْدْ حَمُّو بُو ـ ثْمَارْثْ يَنّا، " آنتش اَسْـلِيغ اَقَّـارَنْ بَلِّي يَلَّا شْرَا ـ ن ـ اُوموكَان

فو غير اِيــنِّي  ـ ن ـ وَذْرَارْ آڨَلّان اَقَّارِينَاسْ لَامَانْ.

لَاعـــوذ  ـ يـوذان ـ ن ـ غَـاذِي ــ  آذافَنْ آمَـك ـ آذ ـ نـدجّــمَـن ـ  ـ آذ ـ عَـاشَـن ـ اِيــذ ـ ن ـ بَــعــضَـاصــن ـ ذِي ـ لَــهــنَـا."
