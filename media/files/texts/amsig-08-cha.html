آمـسيڨـل   اذ ـ اثزيزوتّْ اَسّـعـدّان  اِيض ـ غَرْ آيث حَـمُّـو اِيسـنِيـنَا

اُو ذي ـ اثنزّاڨ زيك  اَطّْـفَــن آبـرِيذ اَنّْـسن كمّلن.

يلّا الحَــال يَـحمَـا لَاباس وَلِّــيش ن الظَّــلّْ

اقْـلي ـ هَاه غَــبَّن.

اَمبــصّح كَـمّـلَن آبريذ اَنّــسَنْ،

اَمـغِـيرْ اَژڔيــن غَـرْ لَبْــعيـتّْ آخّام ذامــزْيَانْ.

واايّا يوعاْ ذِيــسَنْ الرُّوح  اَلّان اَسّارامَن آذافَنْ اقْــلِي ـ ن ـ الرَّاحَثْ.

كُـلّْ مَا قَرّبــَن،

اَسّالَــن  اسْ ـ ذاخَــلْ الحَــسّْ ذامُــقران ـ ن ـ وَخْــباظ ـ ن ـ اُومارطو.

آمـسيڨـل اِيــلَاغا اَسْ ـ نَــزَّهْ، اَمبَــصَّحْ حّدّْ ـاوْـ لَا يـسّْلاسْ.

اثزيزوتّْ اَثـطَبْـطَبْ ـ ف ـ الْبابْ، لامكنّاتي حـدّْ لَا يُــوسِــيدْ.

ذوزّيــنـيْثْ قِــتشّْ،  واَسّـجْـبَانْ ذاخَلْ.

اَژڕيـنْ سَــنّْ ـ ن ـ اِيْرْڨــازَنْ اَطّـڕاقَنْ ذڨ ـ اُوزّالْ ـ س ـ اُومارطو يِـيـژاڨْ.

اَتّْـحَــانُوتّْ ـ ن ـ اُوحّـــدّادْ، اّتّابَركانتْ ذَاسْلُــو،  اَسْ ـ يِـيْـلْـمَـسْ ذازُڨَّاغ

اَسّـيسْ اَتّـافّاڨَـنْ اِيْـصَـفْـضَاوَنْ ذْڨ مُوكَانَنْ ڨاعْ.

اُوذْفَــن واَقِّـيمَنْ ذِيـينِـينْ اَتْـفُورِّيجَنْ ذڨ ـ اِيَّـا ـ آڨّ ـ خَــدّمَــن ـ اُوزَّالْ.

اثزيزوتّْ اَثْ ـ غَـبّْ ـ اَمغِـيرْ هَوِّيتْ الْغومَّـثْ ذڨ اَمكان.

ينّا اُومَاس ـ ذڨّْ ـ اُولْ اَنـّسْ: " عُوضنَـاهِي اتّـاداّرث ـ ن ـ اِينَــمِّـيـتّأ ـ ذايّا ،"

" اَنـلاغَـا  ـ غَرْ ـ الْبابْ، اَمبصّحْ حَــدّْ لَا يَـسْلَانَغْ.

اَن ـ طَبْطَبْ، اَمبَصَّحْ حَـدّْ لَأ يَـسِّـيذفانَغْ.

اُو ـ سُڨَّـأ نُـوذَفْ، نُـوفا اِيـمَانَـغْ  ذِي ـ ثَـالّاسأت ـ اُو ـ ذِيْ ـ الْعافيـفَـث اَثْـكَـنَّـفْ

اَس ـ تِـيِّيثا ـ  ن ـ الْمارطو ـ يَـقْـوانْـتْ ـ آتّيـنِيـذْ آنـشنِـيْنْ نَتْـعَاش ذُڨْ عَژڔ ـ ن ـ اُونِـيْلْ."

فُـو نَڨّارو  ـ اِيَّـا ـ آڨّ ـ خَــدّمَــن ـ اُوزَّالْ ـ اَسَّـرْسَنْ ثِـيغَاوْسِـيـوِينْ واَسّـنْـعْـثَنْ سِي ضُــوضَان اَنّْـسَـنْ الْپـونـــكْ

يَـمُّـــوُڨْ ـ سْ ـ وَصغاڕ ـ ن ـ تَزْدايَثْ  آنِـي اَسّْــغِـيمَـنْ اثزيزوتّْ

اُو ـ امبَـعْدْ اَوِّيــنْدْ آمان ن اتسَـسِّي

سُڨّا اَسَّـڕثاحَـنْ اقْـلِي ، آمـسيڨـل اِيعاوَذاسن ثاقصِـيـصْتْ اَنّْـسَ نتَّا اذ ولتماس.

اَذ ـ لَمشَـقَّثْ آڨَـلّان اَسّْـعَـدّانِـيتْ ذُڨّْ اَبْـرِيــذْ.

يسِّــيوَلْ ـ وَاڨّ ـ خَــدّمَــن ـ اُوزَّالْ ويّـنَّـا، " لَعْـبَذ يَـقّـيمْ آمْ ـ اُوـ صُـوڕدِي

يَـتْواصْڕافْ ذَايَّا اَذْ ـ غَاذي، اَبْلَا مَا ـ آذ ـ يَحْـبَسْ اُوـ يَـتْعَدَّا سُڨْ ݒڔطْفاي غَرْ ـ اُوـ ݒڔطْفاي.

لَعمَــرْ اَنَّغ نَسـعَدّايِـيثْ نَتَّــكَّـأرْ سُڨ مُوكَان غَرْ ـ اُوـ مُوكان،

اُورـ نَـتَّـافْـشْ الرّاحَثْ  اخلاصْ ذِي اَدِّيـمُّث آيّأ!"

يُوثْــلَاي اُومَاسْ او ينّا:  اَمْبَـصَّحْ اِيمَزوُورا اَنّـانْ،

لاكد ما هوضيد سڨ اخف ن اُوـ ثعَرقُوبَثْ، سي اثمورث اورـ اثعدِّيذش، اُوْ كُلّْ صوڔدي غَرسْ سَنّْ ـ ن ـ وُوذمَاوَنْ

اِيـڨَـطُّـو ـ ف ـ ثيشتْ ـ ن ـ تَـغـمُـڔث، مشتا ثلا ثغمرث آفك آڨ ايضو،  يَـتْــغِـيْـمَا آمكْ يّــلّا.

لَــعْـبَذْ بَعْـدْ هَامِّـين، نتّا بَعْدْ غَرْسْ سَنّْ ـ ن ـ وُوذمَـاوَنْ

اِيْ ـ ڨَـطُّـو ـ ف ـ ثيشتْ ـ ن ـ الْجِــيهَـثْ، اُوـ يَـتْــغِـيْـمَا آمكْ آڨ ـ اَلّا.

آس ـ ن ـ واها ذ يحبس سي اثڨــطَّاوث آسّْ اِينين آث ذاسّْ  ـ ن ـ لاخرث,

آذـ يافْ اِيمَـانّـسْ يَـا ذِي ـ الْجَـنَّثْ،  يَـا ذي ـ الْعَـافِـيـفْـثْ!" ـ اُو ـ ذِيــنِــن ـ آذ ـ يَــقِّــيـم !"

ينّا آمـسيڨـل:  ثَـسّْـنَم ماتّـا آذْ ـ يَـضْـڔان آسّْ ـ ن ـ لَـحْـسَابْ؟"

يَنّا ـ وَاڨّ ـ خَــدّمَــن ـ اُوزَّالْ: " اِيْ هِيه، اَقَّـارَنْ كُلّْ اِيْــشْـتْ اَسِّـينَغْ  آذـ اِيْـبَدّْ اَزداثْ ـ ن ـ رَبّـي،"

" اُو ـ كُلّْ اِيـشْتْ آذْ ـ يَقڔا اَزّْمَـامْ اَنّـسْ يَجْـمَلْ ـ اَفْ ـ  وَاـ آڨِّــيـڨَا.

اِيّا آڨَلّان اِيــڨيْنْ ثيــغَاوسِــيوِين  اَحْــلَانتْ، آهَن عَاوذَنْــتْ.

اِيّا آڨَلّان اِيــڨيْنْ ثيــغَاوسِــيوِين  فُـوحــنْتْ، آهَن عَاوذَنْــتْ.

اَمبَــعد يُوذان ڨاع آڨلّان هِـڨِيذ  اِيذْسَــنْ اِي ـ نَـحْـشِـيمَنْ

آذَكّْـرَنْ وآشـ حَاسبن ـ فْ ـ وامْك هَلِّــيذْ هَـڨُّـورذْ اِيــذْسَنْ.

آشْ ـ آزْنـــنْ ذِي ـ ثْ ـ ݒاصݣولْـط  وَلِّــيـشّْ ـ آم ـ نــتَّـاثْ ـ ثَـاڨلّان ـ اُورـ ثَـسَّرْكوسْــشَّا اخلاصْ.

اَمْــبَــعْدْ آيثْـبابْ ـ ن ـ اَدّْنُـــوبْ آذَعْـــنَانْ غَر الْعافِــيفَــثْ

اُو اِيّا آڨلّان اِيڨِــينْ آڨ ـ اَتّــخْـسْ رَبّي آذعْـنانْ الجَـنَّثْ.

 آمـسيڨـل يَـضرَن ثوثلايث آيّا ڨاع ذَڨ ـ اِخْف اَنّـــسْ.

ينّا  ـ فُ ـ اونڨَّـارُو، " اَمبَـصَّح، مانْـدْ وَا اَسِّـينَغْ. آڨَــلَّان يَـتِّيڨ ڨاع ـ وَا ـ سْـيَنَّا رَبّي؟"

اِيـخَژڔيـثْ ـ وَاڨّ ـ خَــدّمَــن ـ اُوزَّالْ ـ اُو ـ يَنَّا." هَامَّــأ اَنَّانْ،"

اَمبَصَّحْ وَلِّــيشْ لَاكَدْ حَــدّْ اَسِّــينَـغْ يِــيْـڨَا آڨّ ـ اَخْـــسْ رَبِّــي.

 يَنّأ آمـسيڨـل: " مَـا يَلَّا ـ ذَ ـ اَتَّايّا اَتّامْــسَالتْ،"

" مَــاتِّا آغ ـ انْـيَـضْـڔانْ نشْـنِــنْ آڨَّــلَّانْ اُوـ نَّــنْـجِـيـحْشْ ذِي ـ لَمْــتِيـحَــانْ ـ ن ـ رَبِّي؟"

اَمَّالَا،  آنّــطْـلَبْ سِي رّبِّـي آذ ـ يِــيلِي ذارحــيمْ  اِيــذْنَغْ،"

اِيْ ـ وَاجْــبَدْ ـ وَاڨّ ـ خَــدّمَــن ـ اُوزَّالْ، سْ ـ اُوـ نَــحْـشَـمْ.  

" غَـرْوَنْ لَخْــبَر مَــا يَــلَّا ـ آذْ ـ يِيْــلِي  رَبِّي ذارْحِــيـمْ اِيْــذْنَغْ  ڨاعْ،

اَنِّــيغْ اِيــذْ ـ نَ ـ اَثْـقَـضْعِيــثْ اَسِّــينَغْ بــرْكْ؟"

يَنّا آمـسيڨـل: " مَانِــي اَغْرِي لَخْبَـرْ؟" ينّا ـ وَاڨّ ـ خَــدّمَــن ـ اُوزَّالْ. " نتّأ وَحــذَسْ آڨلَّان غَرْسْ لَخْبَرْ!"

 ينّا آمـسيڨـل: "  اَثغاوسا آيّا اُوـ ثَسّْــهَـنّايْـشْ اخْلاص!"

اُورـ نتـنَــدجّامش اخلاص آنَّـسَّنْ بَـلِّي آنّـعْنَا غَرْ الْجَــنّـثْ اَنِّــيغْ غَرْ وَمْكَانْ اِيــشْتْ؟

 ينّا ـ وَاڨّ ـ خَــدّمَــن ـ اُوزَّالْ: " اِيْ هِيه، واڨـلّان يَتَّـاڨّأذْ سِي ـ لَـعْـذاَبْ يّــزّمْ فَـلّاس آذ ـ اِيْــسِّـيْـرْمْ آس ـ اَتّْــوَاوْشْ الرّحْــمثْ،"

سُڨّا يّـسْلا ثوثلايْثْ آيّا، يَسِّــيْوْلْ اُومَاس اُو ـ يَنَّا

" اَقَّارَنْ بَلّي الْحَـاسَانْتْ هيشْت بَـرْكْ هَتَّــازَنْ اَمْــيَا سِي ـ هِيَّا ڨَلَّان اُورـ ذَحْــلِينْــتْشَا.

اَمّاَلَا يَــزَّمْ آذ ـ يَتـْواڨّْ لاباس ـ ن ـ الْـحَاسَاناثْ يَحْـلَانْـتْ بَاشْ آذْ ـ يِـيْلِي الْميزَانْ يَـحْلَا."

يُوقَّلْ غَرسْ  اُومَاسْ اَمْـليحْ اَزْدَاث ـ ن ـ وَا ـ هَاذيِــينـِي،

" اَمَّاغَفْ يَـتْـباناكَدْ آتّْـنَدجّـمَذْ  آتّ ـ بَدّذْ اَزْداث ـ ن ـ رَبِّي

باشْ آتَّـازْنَــذْ الشَرّْ   آڨَّـلّان هِيــڨِـيتّْ اِيـذ ـ ن ـ الْحَاساناثْ؟

اُورـ ثَـتْـنَـحْـشامَـذْشْ ـ ف ـ الدّنُوباثْ  ذ ـ تْغاوْسِـسـوِينْ ثيـفَـوَّاحــين آڨَـلّان هِــيڨِــيذْهَنْتْ؟"

ينّا: " آكُـمْـدْ عَاوْذَغْ  اقْـلِي ـ ن ـ تَـقْصِــيْـصْتْ،

اَسْــلِيْــغْـتْ س غَرْ ـ اِيْـمَـزْوُورَا

ثيلّا زيك اشْرَا ـ ن ـ اُ ـ ڨَـلِّـيذْ  ذارڨَازْ  يحلان، اُورـ ذيــتَتّْـشَا اخلاص آكسوم ن اِيــلْفْ.

اُوـ يَلَّا غَـرْسْ ذِي ـ لَقْـصَـرْ واڨَّـلَان  يَـتْـنَاوَلَاسْ الـمَاكَلْـتْ كُلّْ آسّْ.

اَشْرا ـ ن ـ وَاسّْ وايّا آڨ ـ اَتْـنَاوَالَنْ يِــيْڨَا ثيمَرقَـطّْ  ـاِيْ ـ اُوڨَلِّـيـذْ.

يِـيڨَا ذِيــسْ آكسوم ن اُعَــجْـمِي، \ اِيْــزْمَــرْ ـ اَذْ يِــيْـلْف...

اَمبَعْــدْ يرْنِي يَڨْـرا ذيسْ  ڨاع لُـفَّـاحـاثْ ّ ذ ـ الْمَلْحْ ذَ ـ الْخُـضْڔَثْ؛

يَـسُّورْ كُلـْشِي  اُ ـ يَقِّـيمْ يَطْـڔاجَا باشْ آتُّــمّْ اَمْـلِيحْ.

ثَلّا اَرِّيْــحْـثْ اَتَّـازْكِيــمْتْ هَـتّْـشَاث وآڨلّــيذْ يلّا يطْـڔاجَا  باش آذ ـ يتشّْ اَمـلِيحْ.

سُـڨَّا يسّرْسْ واڨّْ ـ اَتْـناوالَنْ المَاكَلْتْ،  آڨَلِّــيذْ ينَّأ، ’ ماتّا ثَـحْـلَا اِيْــمَـڔقَــطّْ آيَّأ!

مَأتَّأ ثِـيـڨِـيـذْ ذِيْـــسْ؟‘

’ اَقْـلِي ن وكسوم ـ ن ـ اُوعَجْــمِـي، اَقْـلِـي ـ ن ـ يَزْمَرْ  ذ ـ اقْـلِي ـ ن ـ يِـيْـلْـفْ.،

يَـكّـرْ آڨَـلِّـيذْ، اِيْـغُــوشّْ. يَـلْـغَضْ فَلّاس ويَنّاس ’شكّْ ذِي ـ اَمطَــنّه ‘

’ اَسْـخَابْـلَاك :آكسوم  ن اُوعَـجْمِـي  اَذْ وا ـ ن ـ يَزْمَرْ آذوعَان اِيـلَفْ يَـحْـلَا؟

سَ ـ الْعَكْسْ، آثْ ـ آكسوم ـ ن ـ ِ يْلْـفْ آڨّلّان  اِيْــفِـيَّا ثِـيْــمَـڔقَــطّْ ڨَاعْ!‘

يَّـمَّـضْـڔانْ غَرْ وا ـ ڨَّـتْـنَاوَالَنْ اُوْ ـ يَنَّا، اَبِّـيْــتّْ اِخْـفْ ـ ن ـ الْــجَاهَلْ آيَّأ.

آبِّـيْتَّاسْ اِخْــفْ هَـڨْـرمْيـتْ  ذُڨْ ـ هَـمَّـاجْ آبَـرْكَانْ، اَسْ ـ نَتَّأ ـ سْ ـ اتّْــمَـڔقَــطّْ ـ اَنَّـــسْ!

آث ـ يـسِّـيْـذْفَــدْ اَغْــرِي لَـفْـسَاذْ  آث اُورـ ذـ اِيسْـتَاهِيـلْشْ غِــير ثَامَتَّانْــتْ!"

اِيْــكَــمَّلْ ـ وَاڨّ ـ خَــدّمَــن ـ اُوزَّالْ ثُـوثلَايثْ اَنَّــسْ اُوـ يَــنَّأ، " آثايَا ماتّا آذ ـ يَضْڔانْ  اِي وَاڨَّلَّان يَـتْـبَـدَّا اَزدَاثْ ـ ن ـ رَبِّي،

واڨَلَّان يَــسَّاذَافْ  غَــرْسْ الشَّـرّْ يّــسُّورْ اِيْـــذْ ـ ن ـ الْخِـــيرْ.

اُوـ مَاتَّأ آسْ ـ يِــيـنِي رَبِّــي اِيْــمَرْذِيــنْـكْ؟ آثْ آسْ ـ يِـــيْـنِيْ،

’ طَّــيّْـشَـتْ آنَـغْــبُونْ آيَّأ ذِيْ ـ الْـعَافِـيْــفْثْ  آثْ ـ يَـسِّــيذْفَــدْ  اغْــرِي لَــفْــسَاذْ!

آث ـ اُورـ يَــسْــتاهِيلْ غِــيرْ الْـعَافِيــفْـثْ!‘  

واَدجّــيْـمَايَا آكُمْ ـ اِيْــنِيغْ بَلِّي ثِـيْـغَأوْسِـيوِيــن  ـ آڨَـلَّان اَحْلَانْتْ  اُورـ تَّـكّْـسَـنْـتْـشَا ثِـيْـغَأوْسِـيوِيــن آڨلّان اُورـ اَحْـلِـيْـنْـتْـشَا،

خَاطْراشْ آڨلّان  اُورـ حْـلِـيْـنْـتْـشَا فِــيَّانْــتْ لَعْــبَذْ ڨَاعْ.

 آنّــجْــبَذْ سَڨّْ اِيْــلْــمَــسّْ  آن ـ اَڨْـرَا ذيس بَــعْــدْ!"

يَنّا آمـسيڨـل، س ـ الصُّوطْ يَــذْلَا:

 آمَاسْ، وَايّا ڨَاعْ يَتَّــدجَّا  زَاوْ ـ ن ـ يّـخْــفْ يَـتْـبَـدَّا!

" يَالّأ اُورـ نَــتْـندجّامش  آنّافْ آمْـكْ آنّــكَّـسْ

آڨَلّان يَــدجَّانَغْ اُوـ نَّــتْــوَاقْــبَالْشا  زْدّاثْ ـ ن ـ رَبِّي ويَـتِّيـيـڨَانَغْ آنَحْـشِيمْ ذي ـ ثِـيطّاوين اَنَّغْ،

زْدَاثْ ـ ن ـ وـ آنّْــبَــدّْ  غَــرْ ـ رَبِّـــيْ؟

اُورـ نَتْــدجَّــامشْ يَــعْــنِــي آنَّــسَّــيـنْـضّــفْ اِيْــمَانَّغْ سِـي اَدّْنُــوبْ  زْداثْ ـ ن ـ وَاهَا ـ اَتْــواڨْــرَنْ ـ ف ـ الْـمِيــزَانْ؟"

" لُوكَــانْ بَرْكْ لَعْـبَذْ آذْ ـ اِيْــنَـدْجَّـمْ  آذْ ـ يَــسَّن بْلَامَا آذـ اِيــشُوكّْ

مَــا يَلَّا  آذ ْـ يِــيْـلِيْ رَبِّـي اِيْــذَسْ ذَارْحِــيْمْ!"  يَنّا ـ وَاڨّ ـ خَــدّمَــن ـ اُوزَّالْ.

" لُــوكَان بَرْكْ آنَّــدْجَّــمْ آنّــمْـسَـفْــهَـام  اِيْــذَسْ اَزْداثْ  ـ ن ـ السَّاعَـث ـ نَ ـ الْـعِـقَابْ!"

سُڨَّا يسْلا آمـسيڨـل ثوثلايْث آيّا،  يَــثّْـنَـهَّــذْ ـ سْ ـ لابَاسْ.

" اَتّايَّا اَتّامــسَالتْ آڨّْ ـ اُوعْــڔنْ  سِيْ ـ ثَــمْــسَالِــيْـنْ ڨَاعْ

آڨَـلَانْ نَــسْـلِـيْـهَـنْـتْ  ذَڨّْ ـ اُوسَّان آيَّأ آڨّْ ـ عَــدَّأنْ،"

" كَـمّْـلَـتّْ آلِـيْــتّْ اِيْــذْ ـ ن ـ تَبْــرِيْــتّْ آيَّأ،" ينّاسَن ـ وَاڨّ ـ خَــدّمَــن ـ اُوزَّالْ،

" وآتّْــخَـلْـضَـمْ سَ  اَطُّولْ  غَرْ ـ اُو ـ فِــيلَاجْ ـ ن ـ لَــهْــنَا.

مَا يَلّا اُورـ اَسِّـيْــنَــنْــشَا نَــهْــنِــيْــنْ لَجْوَابْ، ـــ حَــدّْ اُورـ  اِي ـ هَــتَّــسَّــنْ!"

يُوقّـا الطَّـرْف وِيــس اثْـمَـنْـيا سِي ـ ثَقصيصْتْ
