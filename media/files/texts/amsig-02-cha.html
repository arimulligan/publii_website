كل آسّ يتروحا اَمسيڨل غرـ اَثـْـنَغــڔيث (الكوليج) آڨلّان ثَـتّيلي ذي الجامع.

يلّا يتقاذار الطّالب،  ويتحاوال باش آذي ـ ڨَــمَّــن ف ـ واوالن آڨلّان اتواكثبن ذي ثلويحث،

 اُولامكنّاتي اوذي ـ تْــثَـــبّاتش ايمانّس، اُوريشفّـاش ف ـ واڨ ـ اَلمــذ.

يلّا ديما يسّجباي ف ـ الطّاق.

يَــتْــخمّأم  ذڨ ـ اِيْــضوان آڨ ـ تّيرارن جار ن ـ يِيْحْذِيرن.      

يَــتْــخمّأم ذي ـ زيزوا آڨلَّان ثــتّْـزَنّ ذڨ اقْــرڕيعن.

يَــتْــخمّأم ذڨ جَرْوِن ـ آڨلّان ـ اطقڕقڕن سمان ن ـ تعوينين.

يَــتْــخمّأم ذڨّ ـ اَفڕاخ  آڨلّان اَتغنّـان ذي ث ـ صطّوا سمان ن ـ يغژڕ.

يمّكثي ڨاع واڨّ ـ  اَژڕا ـ ذي ـ ثِــيْــزڨِـيْ وايسَقصا اِيمانّس،

" مُكشا اثغمّيد ثاصطّا  ثعذل سَڨ زوران عَوْجَنْ؟"

يتخمّام ذڨ اِلمَسّْ آڨلّان يمُّوڨْ سِ ـ حْذِيرَنْ ذي مُقرانّن أسمان ن ـ تعوينت ـ اُوـ اِيسقصا اِيمانّسْ،

" امّاغف الدخّان يتّالي وآمان هُڨَّان؟"

يمّكثي ثي ـ عَـدّجَـاجِـيْـنْ آڨلّان اَتكّارنت ذڨ مسّاي اُوْ ـ اِيسقصا اِيمانّسْ،

" مانيس يتّاسد لَهْـوَا؟

ثيسطّوا آڨلّان اتهايالنت لهوا ـ اَنّيغ   ـ لَهوا آڨلّان يتهايال ثيصطّوا؟"

واِيروح سِي خَمّيمَن انّس غر لَبعيتّ، جارْ ن ـ اَتْـــزَڨــوَا  ـ ذَ  ـ اَلْكيفَان،

وآخمّم انّس يوّيث آنِــي  يَــتِّيــلِيْ آسَڨّان اذْ يَثرَان ـ نِـي ـ جَنْوانْ.

اَلّانت لْخَطراث لاباس آني يوغا ذيسنت هيّيثا سوقبّال خاطراش اُورـ يَــسّـيـنشا آوالن آڨلّان ذِي ـ ثلويحَث اَنّس.

شراـ ن ـ لخَطرث ايروّح غر وخّام ويوفا زيزيس يقّيم ـ ذڨ  ـ اِيــمِـي ـ ن ـ الْـبَـاب.

" آزيزي اَلّانت ثيغاوسيوين اُورـ ندجّيمغشّا آهـنـتـَفـهمغ!"

ويوعا فلّاس آشيبان،

"الظلّ ن ـ تزدايث يتّيلي يبعذ ـ ف ـ اِيزوڕان ـ انّس.

كمّل اُورزِي ـ ف ـ لوجاباث، آممّي، امغير آتّافذ  ـ آـ ثَـخْسَـذْ!"

اُوسّان اِنين يمّاس ن ـ اَمسيڨل هطراجا آمتّشوك ويس اثلاثة.

لامكنّاتِـي ثاروَا اَنّس هلّا ثوعــڕ وآبزّاح اِيلولَدْ يَمّوثْ.

اثولّى ثُغلَب امغير اوهتندجّامشا لاكد آتكّر سڨ سامو انّس.

ثَقّيم سنّ نويورن ونتّاث اثعلّق جار ن ـ تمتّانت ذ ـ تّموذري.

آرڨاز اَنّس اِيرَوّحَد سڨ فيلاج ويژڔا الحالت اَنّسْ

اُوثقوّيمش لاكد آتَّـنفَــع ذڨ اَخّام،

يكّر، يلّفاس اُوْ يرني يرشل س ـ تمطّوث هيشت.

اَثروّح غَرْ ـ لُـمَّـالِي ـ اَنَّـس وف ـ ثڨارا ن ـ ثلاثة نُو يورن ثَمّوثْ.

ثامطّوث هاجديتّ ن ـ وخّام اوثْلَتّاش لاباس س ـ اَمسيڨل، اَذ ولتماس اثزيزوتّْ.

آمك اَقّارن اِيمزوورَا،

" اِضارّن نوڨوجيل اسكّارن آغَبّار آلدا آتّيلي هتشّاث النوّث

واَسَّـكَّـارَن اِيلاطّن ذي ثغاڕوڕث نو ـ نَبْذو!"

اَخسَن آذسّرشلَن اثزيزوتّْ  اِي ورڨاز ذڨ فيلاج يلّا يرضَلْ اِيصوڕڊان اِي ذادّاس.

لَعمر اَنّس ولا ثژڕيث ولا هسّن فلّاس لاكد اثغاوسة

نتَّـاث غرس ثلطّاعش ني سڨّاسَن ذايَنْ.

هلّا ثيلّ نزّه نزّه

وآمسيڨـل اوذيوفيش آمك آذيصبّڕ ولتماس،

خاطراش نتّا بعد يلّا يحزَن لاباس.

اقّيمن ربعـَا ـ ن ـ ووسّان ـ ف ـ العرس.

ذڨّ ـ  اِيض اِينين بَعد اوسيند اِعَنّابّن.

اّغزين آهَــمَّــاج ـ ذي الحِيظ ـ ن ـ اتحانوتّ

وَ وِّين ـ آسـنـدُوق ـ ن ـ الصّواڕڊ.

ذي ثنزّايث زيك، آيثباب ن ـ تحانوتّ اَتعدّان سڨّ اَخّام غَرْ وَخّام،

اتسقصان كلّ اِيشت ماني يسّعدّا ايض اِنين.

اوفين بلّي ڨاع اَلّان ذڨ اَخّامن انّسن، غير اَمسيڨل.

يرڨ ايض اينين باش آذيعوسّْ اِيثران آڨلّان ـ اَتعدّان ذڨ جّنّا اُو خسّين؛

يخسْ آذيژڔ ما يلّا ڨَطُّون ـ ذي ـ ثِــيْــزڨِـيْ  

انّيغ ف  ـ اُوـ غيـڕ ـ أِينّي ن ـ وَذرارْ.

ڨيذَر اِيعاوذاسَن ڨاع

بلّي يژڕا اَمسيڨل  يڨور ف ـ التّيم ن ـ تحانوتّ.

روحن سْ لَعْزَمْ اسّكرند آمسيڨـل سُڨ سامو اَنّسْ.

طــيــشِـيـنيثْ ذُڨ همّاج يغضا

واَدّجينيث ذينين بلاشي الماكلت  ـ ذ ـ بلاشي ن ـ وامان.

آمسيڨل اوغرسّشا لخبر ڨاع اسْوَا ـ اَڨَّضران.

يلّا يسّال الدّراري اِينّي  حفّظن ذاخل ن ـ الجامع،

امبصّح حَدّ ـ اُوديوسي غرس آثيژڔ.

يَقّيم ذينين سنّ ن ـ وُوسّان اَذْ سَنّْ ، يِيْضانْ،

اِيخَمّيمن عرّمَن اسّورَن ذڨّ اِخْفْ اَنّسْ.

اَمبعد آقضعيث ن ـ يرڨازن سُڨ فيلاج اُوسيند واسّرڨينيثيث سڨ هَمّاج

واَبّذان اتشّاثن ذيس،  سوقبّال واَقّاريناس،

" هاثايا آڨّستاهل اُوـ عَنّاب!"

اَمبعد حاوزينيث سُڨ فيلاج،

حجّڕيــنِـيـثْ ـ سِي ـ مَسّْلا غَرْ ـ اثْرعقاي.

يولي اَمسيڨل ـ ف ـ ثعرقوبث ف ثبريذتّ آڨتّاوين غر ـ ثِــيْــزڨِـيْ.

يوڨير امغير يُغلَب اُوذيندّجيمشا آذيرني غر ـ زداث.

هوّيث الغُومَّث اُو يُوضي زدّو ـ ن ـ تصطّا ن ـ لْكَرْمُوصْ. (اِيْـمَـطّـشَّـان)

سُڨّا يلّا يطّصْ، آوال ن ـ تمتشّوكث يسّربزاث اُو يكّر.

يّمّضران اُو يَعْـقَـلْ اثزيزوتّْ

هوزّلْدْ غَرْسْ

ماتّا ني ـ ذامّن آيّا ڨاع ف ـ وُوذَم آَنّك؟"

" آشم هتّاقّالد،" يَنّا،

"امبصّح ـ ف ـ وَآڨ ـ اَضــڔان، اُورـ ضَــلّــغـش ـ ڨاع!"

ثطّفيث سُڨ فوس اَنّسْ اُو ـ ثنّا:  " آرواح، آنّاذف غر ـ وَخّام!"

" آراح!" ينّاسْ. لُــمَّـالِـي أُوـ يَّـخـسّــنــشا،

لاكَــد آفيلاج أوـ أِيخسشَّا

حاحينايا اَسْ يَحذيرَن!

آذ ـ روحغ آذورزيغ فوموكان ـ ذيـس ـ لَـهـنـا اُوـ يحلا

 ثنّاس اثزيزوتّْ: اَمّالا كيعَاد هامّا، آداسغ اِيذَكْ!.

" آراح، آبريذ آيّا يوعَرْ لاباس فَلّام،"

"اُورـ ثَــتـنـدجّامذش."

" مَشتا يڨڊڔ، " ثَــقِّــيـم ثَــتّـنـتخالاي فَـلَّاس

" ماتّا آيَقّيمن ذڨّ اَخّام غير الْخَذْمّث نوڨڔصّنْ، اَذْ لوعاڔث انّسن فلّا؟"

" ذي الْوَقّث اِينين بعد ـ اَسْلِين آمُوـ وَشوَشْ ـ جار ـ ن ـ تِـيـوْرِيـقـين ـ ن ـ اَتـسطّا ن ـ لْكَرمُوصْ (اِيْـمَـتـشَّـان):

" اُورزِيتّ آتّافم.

اُورزِيتّ آتّافم."

اَمّضرانَن وَ اَڗڔين زيزيثسَن يَتوارّذدْ غَرسَنْ.

" آثاين، زيزي،" ثَنّا تازويت،

ثَنّا اثزيزوتّْ: "ذ اتـشَّـكّ آڨَّـنَّـان آمَّـا؟"

ينّا: "أورـ اَنّيغ لاكد قْلي، أثا معزوزت اِينوغ‘‘

يسّيوَل اَمسيڨـل: " اَمّالا، اَمّاغف ثوسيدّ غَر ثِــيْــزڨِـيْ، آزيزي؟"

اِيواجبيث:" روح إيمير،  اَيا آمسيڨـل!"

" ربّي يخساشّ باش  آتّورزيذ وآتّافذ.

آقّل اِيْ الدِّيمُّث آيّا آمك اَتّامُقرانتْ!

روح، اُورزي ذڨ موكانن ڨاع، اَمغير آتّافذ ثـيـذَتّْ هجْمَلْ

اَمغير آتسّنذ كّلّش آمك اِي ـ ثلّا ـ ثـيـذَتّْ

وآت ـ اَسّـنـذ ـ آڨ ـ اَتْ ـ وَافّرَن فلّانغ."

ثوثلاي اثزيزوتّْ:

ثنَّـا:" آنتش آذروحغ اِيذَس آزيزي!"

" روح، لاكد شمّ، آثامعزوزت اِينوغ،" يوعا فلّاس.

" أُومام ـ آذ ـ ينطهلّا ـ ذيم.

آنتش اسّنغ واڨلّان يَخْسْ  آذ ـ يرشل اِيسّم

اِيسَسّْ الشّرابْ. آتّـيلـيذ اثـبَعْـذَذْ خِيرام."

وامْبَعد يَقّيم آمغار يتّاقّال غَرسَنْ.

" اَسّغذَتّْ!" ينّا. "ثلّا اثغاوسة هيشت يزّم آكمتينيغ.

اُورـ آسّْ ـ  نَـزْڨـامتّـش ـ ف ـ ثغاوسة آڨ  ـ اّرسن ـ ف ـ اَدّمّـث ـ انّون.

آتْ ـ ذي ـ لَحْظا اَمغير آتولّامْـدْ!"

واَمبعْدْ يرفذ فوسْ أنّسْ

وآمسيڨـل ذْ اَتـزِيْـزُوتّ اَسّوذنينيث ف يَخْفْ اَنّسْ.

اِيبَدّ ويوڨير  ـ غر ـ اُوفيلاج.

آمسيڨل اَذْ اَتـزِيْـزُوتّ اَطّفن ثابريذتّ.

" ماتّا يَخْسْ آذييني زيزي،" اَهسقصا اِيمانّسْ،

" سُڨّا يوثلاي ف ـ ثغاوسة آڨّرسَنْ ـ ف ـ اَدّمّـث ـ اَنّغ؟"

" ثلّا اَثغاوسة يَفّرِيتْ،" يوعا فلّاس اُوماسْ،

" امبصّح اُورسّينغ ماتّا ثلّا.

يَسّن اَثغاوسة،  نشنين اُوتنسّينش."
