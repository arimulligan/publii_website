آمـسيڨـل اذْ ثيزيزوتّ عـدّان اُوڨيرن اَمغِير النّصاف ـ نْ ـ وَاسّْ؛

 وآبريذ يلّا يثّلُوّاي جار ـ ن ـ تصطّوا.

اُوْ غَبَّنْ وَ اَلُّوژَنْ.

اَژرِين ثامغَارْثْ ثَتْلَايَام ذِي لَحْشِيشْ.

" اَثْعَيّضاس ثيزيزوتّ: " آماَسَا حَنِّي، اُورْ ـ غْرَمْشَا اقْلِي ـ ن ـ وَامَان ـ ن ـ تْسَسِي؟"

" آرْواحَتّيتّْ ذَايّا، " ثَنّاسّنْ، " آكمَاوِيْغْ غَرْـ ثَعْوِينْتْ."

تَبّعِينيتْ، اوْ عَدّان ف ـ قْلِي ـ ن ـ اتْڨَربِيثْ

ثَتوَاخْذم اسْ ت تَمقِيصبث ذَ الطّالع ـ ن ـ الدّفلا آوڕْديث، وُـ اُو لِينْ ف ـ شا ـ ن ـ وَبرِيذْ

اَمغِيرْ خَلْضَنْ غَرْـ ثَعْوِينْتْ ـ ن ـ وَامَان صَمْڊَنْ.

سُڨّا اَسْوِينْ امْغِير اَرْۈان، ضَلْقَن اِيمَانّسَنْ ف ـ اثْمُورْثْ باشْ آذَسَّڕثاحّنْ.

اُوـ ذيعَدِّيشا قيتشّ ـ ن ـ الزّمان، اَمغير يَوّيهَنْ اُونودّم.

اَقْلِي هَاهْ آمـسيڨـل يَكّر سُڨّأ يَسْلَا اِيْ ثَمغَارثْ هِيلّ.

يَسكّر وَلْتمَاسْ اُو سَقْصَانِيتْ.

" ماتّا آشميوغَن آ مَاسَا؟ اَمّاغَف ثِيلّذْ؟"

ثَطّفيت اَثزيزوَتّْ ف ـ اُوفوسْ اَنّسْ اُو ـ ثَنّاسْ، ماتّا آڨّضرَانْ آمّاسَا؟"

" آثڨڊَڔْ اَدِّيمُّثْ آيّا!" ثنّا.

" آكُمْ اُورـ ثتندجّامَمْش  آتّژڕَم آمك آثَڨݚڕ

اَزداث ـ ن ـ وَاآتّعاشم ذيس!" اَث ـ شِيفّيهَنْ.

" لِيلّْشا، آ مَاسَا!" اَنّينَاسْ.

" عاوذانَغْ آڨَّضران."

" يا ولَايْلّه فَلّا!" ثَنّا ثامغَاڕث " اُورغريشْ لاكَدْ آمَدُّوكَلْ ذِي اَدِّيْـمُّث آيّا!

حّدّْ اوْ لَا يَتخمَّام فَلّا اِيمِيرَا نَتشّ اَتّامْغَاڕْثْ.

سِيمَا اَلّيغ اَتّامَزْيَانْتْ اُو نتشّ اَتعَاوَاذغْ اِينزڨّامْ اِينُوغ اِيْ رَبّي،

واِيمِيرا سُڨَّا يُوذانْ ڨّاع  اَدجّينَايَا، نَتّا بَعْدْ يَدجّايَا

اُورـ يَسّالْـش اخْلاص. اُور يَدجّي غِيرْ اِينَزڨّام اَذْ لَحْزَنْ."

ثَـنّا تَازويتْ: مَشتّا اِيِّينّي اَتُّونْ،

نَشنِين اُور ـ نتَّتّوُش اخلاص اَثمُوحلِي اَنّمْ اِيذْنَغْ".

ثنّا ثامغاڕث: " ثَژڕِيم اِيلْمَسّْ آيّا ذَ ـ لْعَافِيفْثْ؟"  

اِيصَفْضَاوَنْ اَتَافَاڨَّنْ اغْرَنَّجْ واِيصْغاڕَنْ فحْمَن اَتْرُوسَنْ.

لَخْذَأمِي يَحْلَانْتْ آڨّلّان نَتّيڨّْ اَتّافّاڨَّنْتْ اُورـ اِيتْغِيمَا غِيرْ اِيغَذْ."

" اَمبَصّخ لَخذَامِي يَحْلَانْتْ اُورـ اَتَّفّاڨَّنْتْشَا غَرْ يِيلُّو؟"

اَثْسقصَا اَثزيزوَتّ. " مَانْدْ وَا آڨّلّان غرسْ لَخْبَر؟"

ثنّا ثامغاڕث: " مَاندْ وَا آڨّسّنَنْ بلِّي اَثخَلّض ربِّي اَنِّيغ  آرَاحْ؟ آنشنين ـ ذڨّ ـ اِيغَذْ ـ

ثايّا آنتش اَتِّي ـ اَمثبّتّ اَسِّيس."

ثنّا اَثزيزوَتّ:" بَركَا سَڨّْ ـ مَطّاوَنْ، آماسا!"

" عاوذانغ بَرك ف ـ اِينزڨّام اَنّم

باش لاعود آندجّم  آشمنعاون.

ثَبْحَر ثامغاڕثْ اُوـ ثنَّـا: " يعني ثخسَم آتسّنم الصحّ؟\"

اَمبَعْد ثخژڕْ ذِيسَنْ  اِيسْنِينَا اُو ـ ثَنّا:

ربي آكُمْيَصْطَڕ سَڨّْ ـ اَعْذاوَنْ آمْ اِيّا ڨّلّان اُوسِيندْ فَرْقَن جاراثنَغْ نتشّ اَذْ وَرڨّاز اِينُوغْ!

تَهمِينَايَا س ـ اَتْغاوسة لَعْمَر اِينُوغ لَا اِيڨِّيغْتْ اِيحَاوْزَايَا سَڨّْ ـ اَخّام،

اّكّرَغ رَوّحغ غَرْ لُـمَّـالِي اينــوغ.

لامكنّاتي سُڨّا خَلضغ، اُوفيغهَنْ اَمُّوثَنْ ڨّاعْ.

آخّام اَنّغ يَتْوَاسْدغْ اسْ ـ يُوذان اُوهَنَّسِّينشا، اَوِّينِيث اَذ ـ غِيلْ.

ثُوّايَا اَلْغُومَّثْ  سُڨّا وَلِّيغْدْ غَرْ اِيمانِ ينُوغْ،

اُوفيغ بَلِّي اُوكرِينَايَا سَنّ اِيصوڕڊَان آيّقّيمَنْ.

اَرْنِيغ اُوضِيغ لَخْطَرث هِيِشْتْ ف ـ اثْمُورثْ، اُور ـ اَسِّينَغْ آهاذيڨَّغ اَنِّيغ مَانِي آذ ـ قَدَّغْ.

اَوِّينايَا ڨّاع ـ آـ غْرِي، آرڨّاز اِينُوغ، ثَاروَا اِينُوغْ، آخّام اِينُوغْ، اِيصوڔڊَانْ اِينُوغْ.

آقّلَتّْ آمَكْ اِيصَابَايَا رَبِّي! آنتشايَنْ ذِي ـ ثِــيْــزڨِـيْ (الغابة)

جَارْ ـ ن ـ لُوحُوشْ اِيْـبَرِّيَّنْ، اَنّضَنْ اِيسِّي اَشّواطَنْ.

كلّ آسّْ، اَتْلَايَامَغ لَحْشاوَشْ،  ذتِّيوْرِيقِينْ ـ ن ـ ثِــيْــزڨِـيْ. (الغابة)

اَقْلِي تتّغثْ، وَ اقْلِي  خدّمَغْ اِيسّْ اّدّْوَاوَاث باش آصّڔثاحَن اِيموضان.

اُو كَنوِينْ، مَانِي آتّعْنَام؟  بايَن اَثبَعْذَمْ ف ـ لُمّالِي اَنُّونْ؟"

ثنّا اَثزيزوَتّ: " آنشنين اَنرُوزّي فُ ـ اُوموكان ـ ن ـ السّلامت اّذ ـ لَهْنَا، آمّاسَـا"

" اَمّالَا ثَتحَوّاسَمْ  ف ـ لْجَنّث" ثنّا ثامغاڔثْ.

" اُورـ هَتّافَم  لَا لَهنَا اُو لَا السّلامت ذايّْ ذي اَدّيــمُّث آيَّـا.

آقّلَتّْ، ڨّاع آڨَّدّرَنْ يژلْ اِيغَالَّن اَنّسْ غَرْ ـ اُوجَنّا،

اَخْسَنْ  نَزّهْ آذخلضّن الجنَّثْ.

لَحْشيشْ يَتراڨَّدْ سِي ـ ثْمُورْثْ؛ ايغونام اَترا ڨَــند سُڨْ شال، ثيصَطّوَا اتّالينت غرـ اُوجنّـا

الدّراري اَتڨَعْمَارَنْ؛  اِيغَطَّنْ اِيبَرِّيَّنْ اَتَّالين غَر اَثْڨَعْمَارْ،

وِ ـ اِيفْرَاخْ اَتّافّاڨَنْ زَنْڨْ ـ ن ـ كُلّشْ.

كُلّْ اِيشْتْ يَتَّخْسْ آذ ـ اِيوَارَّذْ  غَر سيدي ربِّي

اَمبصّح مَانْدْ وَا اَسِّيسَن آڨلأّان آذ ـ اِيندجّم آذ ـ يَخْلَضْ غر ـ وَخّام اَنّسْ، الجنّث؟

آقّلَتّْ آها ـ سَنْيَضرَان ڨاع. ڨاع واڨلّان يخْسْ آذ ـ يَالِي

آذيتواوعا غَرْـ وَادّا  سُوـ غِيل. لَحشيش آذ ـ يَلُّوتَّمْ، اِيغونام آذ ـ عوجن،

ثيصطّوا آضاضونت؛ اِيغطّن  اِيبَرِّيَّنْ آدهوان غرـ اِيغَژڔَانْ؛

اِيفْرَاخْ آدولّان آشنڨْرَنْ ف ـ ايعوراف واَدّرَارِي آذَمّْثَنْ.

آذولّان ڨاع  غرـ ثمُورثْ آنِيسْ آدوُسِينْ.

ولِّيش ـ ن ـ واڨلّان يَـخـلض غَرـ اَغْرَنّجْ انّيغ يـسـعَـا  لَهنَا آڨلّان يَخسِيث نَزَّه."

  آمـسيڨـل يَنّا: " آنتش اَسْلِيغ اَقّارْنْ،"

" بَلّي اِيرْڨازَنْ ڨاع ذي ـ لْجَنّثْ آذيلينْ آمْ اِيڨَلِّيذَنْ ذِي مُقْرَانَنْ."

سُڨّا هَسْلَاس تَازويت ثَخْژڕْ غَرْسْ لخژرث ـ ن ـ  نُوسَقصِي.

 اُو ـ ثَنّا" ااُوْ مَاتّا فلّانَغ نْشَنْتِينْ، الْخالَاثْ؟"

" اُورـ نَتِّيليشْ  ذِي ـ الْجَنَّثْ، نْشَنْتِينْ بَعْدْ؟"

ثَرنِي ثُوثْلَاي ثامغاڕث لَخْطَرثْ هِيشْتْ اُوـ ثنَّـا

" اُو ـ نَتّأذَفْشْ لَاكَدْ غَرْ ـ الْجَامَعْ،"

" آمك آندجّم آنّاذَفْ غَرْ الْجَنّث؟

نْشَنْتِينْ اَلْخالاثْ اُونَّسِّينشْ لاكدْ آنّژّالّْ

اونفهّمش   ماتّا نقّار ذي اث ـ زالّيط.

آوـ ثنا:  اُوغْنِيرُوحْشَا ذِي ـ اثْغَاوسِيوِينآيّـا"

" آليلّشا، آمَاسَا!" ثَنتْخَالْ فَلّاس اَثزيزوَتّ.

" يِيلُّو آذ ـ يِيلي ذاـ رحيم اِيذنَغ،  ما يَلّا ا َذْ ـ وَا ـ آڨ ـ اَخسْ."

هَنّيخْسَسْ ثامغاڔث اوـ ثنا:" لامكنّاتي مُكْشا آندجّم آنّسّنْ،"

" مَا يَلّا اَذْ ـ وَا ـ آڨخس اَنّيغ آرَاحْ؟"

 ثَنّا اَثزيزوَتّ:" اَقّارن بَلّي رَبِي ذا ـ رْحِيمْ،"

 ثنّا ثامغاڔث:" منهو آڨْ ـ ضَلّنْ؟"

" مَنهو آڨ ـ ضَلَّنْ بَلّي آذ ـ يِيلِي ذارحِيم، فَلّا، اَنِّيغ فَلّام، اَنِّيغ فْ ـ اُوْمَامْ؟

لُوكان يَلّا بَرْكْ واها ذْ يَكّسَن ف ـ اُوُلاوَنْ اَنّغ

هِـيـوْذِي ـ ن ـ اتْغاوسِيوِينْ آڨلّان ذِي   ـ لَاخَــڔثْ!"

ينّا آمـسيڨـل: " آمَنْ بَرْكْ اسْ رَبِّي، آمَاسَا، اَثْ ـ لَاوِيذْ ـ سَ ـ اَصّْبَڔ،"

 ثنّا ثامغاڔث: " كم ـ ن ـ لْخَطْرَاث آسْلِيغ  ثوثلايين آيّا ،"

" اَمبَصّح آتْ اَزِّيْ ن ـ اُويور اِي واڨلّان يڨّور ذي ثالّاسْتْ

اِيْرَنِّي اُو يَتَّنْقِيصْ، يَتڨَعْمَر اُوْ يَطَّمْژي، يتروحا اُوْ يَتّاسَدْ،

اُ حَدّْ مَا اِيضَلْ بَلّي آس ـ يِيڨْ اَث ـ فاوَتّْ اَمغِيرْ آخَّام اَنّسْ.

نتڕَاجا باش أنّسْلْ اَثغاوسا اتّاصحيحَثْ،

اَثغاوسا اَزِّي ـ نَ ـ تْفوكَثْ  ثتلَالَا آسّْ يَجْمَلْ

اُور ـ يحببّسشا سُڨ ـ لَالِي ـ ن ـ وَبْرِيذْ آڨلّان آغ ـ اَنْيَاوِي غَرْ ـ اُوموكانْ آڨلّان نَخْسِيثْ.

آه، ذِي نغبان آنَلّا! مَنهو آڨلّان آذيندجّم آغ ـ نيسّڕثاح سِي ـ ثِيوْذِي آيّا ذَ الشكّ آيّا؟

مَنهو آڨلّان آذيندجّم آغ ـ نِيسَلّكْ  سِي اَڊّرْكْ ذَ ـ لْقَلْقَثْ آيّا؟

مَنهو آڨلّان آذيندجّم آغ ـ اَنْياوي غر ـ اُوموكان ـ ن ـ لَهنَا ذي اَدِّيمُّث آيّا ذَ ـ لَاخرث؟"

ثَنّا اَثزيزوَتّ: " آنشنين اَنْرُوزِّي،  يَنّا اُومَا ذَ اَنّتشّْ،"

" لوكان آنّاف ويشت يَسَن وايّا ڨاع.

لوكان آنّاف،  آدنولّا وآمنِّينِي!"

ثنّا ثامغاڕث: "روحَتّ اَمَّالا"

" حَوّسَتّْ ـ فْ ـ ويشت آڨلّان آذيندجّم آغ ـ اَنياوي  غر ـ اُوموكان ـ ن ـ لَهنَا!"

 ينّا آمـسيڨـل:" نمسفهام، آمَاسا،"

" اَمبصّح زداث ـ ن ـ وانّوڨير، آندجّم آشمنعاون."

واَبذان خَدّمَن اَقضان اَنّفَع ـ ن ـ تْڨَربِيث

اُو فَرضَن آبريذ آڨ ـ اَتّاوِين غَرْ ـ ثَعْوِينْتْ.

اَوّيند آقَضعيث ـ ن ـ اِيمَسْلَا ذي ـ مُقرانَنْ اُوْ خَذميناس اِيلْمَسّْ.

اُوْ اَمْبَعدْ اَسّوذنِينيتْ ـ فْ ـ ثِيمّي اَنّسْ واَرنِين اُوغين آبريذ اَنّسَنْ.

وكمّلَن ثيكلِي اَنّسَن اَمغِيرْ خَلْضَن اِيغژڔْ يَحْمَلْ.

عَرّمَن يُوذان آڨلّان بدّن ـ فْ ـ ثَاغْمُڔثْ اُور ـ ندجّيمنشا آذاغڔَنْ

خاطراش يلّا يتّازّال ـ سَ ـ لَقوث ويُوِّي اِيذَس ثِيمصِيغرثْ.

آقضعيث ـ ن ـ يَرڨازن اُوِّيند  الطّالع ـ ن ـ تصَطّوَا ذي ـ سَغْوَانْ.

اقّنينِيهَنْ غَرْ بَعضَاصَنْ، واَمبعْد كَڔكڔينيهَن

باش آهن ـ صِّيغرن غَر ثَاغْمُرْثْ اِينِّي ـ ن ـ يغژڔْ

اَمبَعْدْ،  ضَوّلَنْ اِيخْفَاوَن اِيسْنِينَا غَر ـ  اِيمَسّْلَا اِيژاڨَنْ

باش آتّيلي ثيمصيغڔث ثڨڊڔ، اُوـ يُوذان آذ ـ عَدّان.

ذِي الوقّث آڨلّان يلّا ذيس آمـسيڨـل اذْ اَثـزِيـزُوتّ  اَطّاغْڕن،

اَسلِين لّعْيَاظ  ذامُقران ورناثْسَنْ.

اَمّضرانَن باش آذزڕَن. اسّاعا اَتّمَطُّوث هَزْلُولَّقْ سِي هَمصيغڕث.

ثَويــتْ لَحْمَلْتْ.

يوذان ڨاع اَتعِيّاضن، واَتّازّالن غَرسْ

اَمبصّح لَحمَلْتْ ثَلّا ثَڨڊَر لَابَاسْ و ثَوِّيتْ اَزداث ـ ن ـ واها تْخلْضَنْ.

آمـسيڨـل اذْ اَثـزِيـزُوتّ   اَثْشِيفّيهَن لَابَاسْ.

ثنّا اَثزيزوَتّ:" مَنهو آڨ ضَلّن مَلمِي  آذيعيّض ربّي ـ ف ـ وِيشتْ اَسّينَغْ؟

" ومنهو آڨ ـ ضَلّن ماني آنّيلي اِيذ  ـ ن ـ ربّي مَا يِيڨيتْ؟"

الوقّث ذامسّاي ونَهنِينْ وَرعاذْ ذي ـ ثِــيْــزڨِـيْ ؛ (الغابة)

واَطّصَن ذينين اَمغِيرْ ثانزّاڨثْ.

يوقّا الطّرف ويس خمسة سِي ثَقصيصتْ
